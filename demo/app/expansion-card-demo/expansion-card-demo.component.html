The <integrity-expansion-card> component is used for displaying expandable cards. 
        <integrity-expansion-card
        title="French Id"
        description="Default ID"
        icon="person_outline"
        [sections]="sections"
        error="false"
        [isDefault]="isDefault"
        (bottomButtonClicked)="onBottomButtonClicked($event)">
        </integrity-expansion-card>
    
        export interface IExpansionCardSection {{ '{' }}
          fields: Array<IExpansionCardSectionField>;
        {{ '}' }}

        export interface IExpansionCardSectionField {{ '{' }}
          name: string;
          values: Array<string>;
        {{ '}' }}
Attribute Required Value Description
title no text Title
description no text Description
icon no text The name of the material design icon
sections no IExpansionCardSection Sections displayed when the card is expanded.
error no text If the value is anything but '' or 'false', the color scheme and the description will turn red.
isDefault no text If the value is anything but '' or 'false', the card is marked as default
bottomButtonClicked no click event Click event triggered when the bottom button is clicked.
