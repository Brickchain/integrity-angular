The ParentService is a utility that allows the admin UI for a controller to call handlers in the realm admin UI.

To use the service start by importing the importing the parent module in the app module.

import {{ '{' }}ParentModule{{ '}' }} from '@brickchain/integrity-angular';
  
    @NgModule({{ '{' }}
      ...
      imports: [ParentModule],
      ...
    {{ '}' }})
    export class ControllerAppModule {{ '{' }} {{ '}' }}

Next inject the service into the component that wants to use it.

import {{ '{' }} Component {{ '}' }} from '@angular/core';
  import {{ '{' }} ParentService {{ '}' }} from '@brickchain/integrity-angular';
  
  @Component(...)
  export class EventsServiceDemoComponent {{ '{' }}
  
    constructor(private parent: ParentService) {{ '{' }} {{ '}' }}
  
  {{ '}' }}
 
updateActions() {{ '{' }}
  this.hassClient.getDescriptors()
    .then(actions => this.parent.postMessage({{ '{' }}
      '@type': 'update-actions',
      'actions': actions,
    {{ '}' }}))
    .then(() => this.notification.displayMessage('general.actions_updated', 'label.close'))
    .catch(error => this.notification.displayError(error));
{{ '}' }}
